Usage Example 

# Python example using requests library
import requests
import json

headers = {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
}

data = {
    'filename': 'example.py',
    'content': 'print("Hello from Nepantla API")'
}

response = requests.post(
    'https://your-api-domain.com/api/write',
    headers=headers,
    data=json.dumps(data)
)

print(response.json())
